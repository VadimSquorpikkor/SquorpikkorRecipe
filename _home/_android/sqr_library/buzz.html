import android.app.Activity;
import android.content.Context;
import android.os.Vibrator;

/**
 * Вибро на действие: можно использовать, например, вместе с OnClickListener:
 * Buzz b = new Buzz(requireActivity());
 * clearList.setOnClickListener (v -> b.buzz(() -> mViewModel.clearList()));
 *
 * v - 1.00
 *
 * */
public class Buzz {

   private static final int VIBE_TIME = 40;
   private final Vibrator vibe;

   public Buzz(Activity activity) {
      vibe = (Vibrator) activity.getSystemService(Context.VIBRATOR_SERVICE);
   }

   public interface Buzzer {
      void doOnBuzz();
   }

   /**Включение вибро. В интерфейсе указываем, что делать*/
   public void buzz(Buzzer buzzer) {
      buzzer.doOnBuzz();
      vibe.vibrate(VIBE_TIME);
   }
}
